def name(summoner_ids)
  neated_param = neat_params(summoner_ids)
  names = get(resource_path_by_id('name', neated_param), @version)
  names.map do |name|
    RiotGamesApi::LOL::Model::Summoner::Name.new(id: name.first, name: name.last)
  end
end
